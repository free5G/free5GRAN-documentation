59 int phy::cell_synchronization(float &received_power) {
70  BOOST_LOG_TRIVIAL(trace) << "PSS synchronization";

77  size_t num_samples = 2 * ssb_period * rf_device->getSampleRate();
78 
79  // Create buffer
80  vector<complex<float>> buff_2_ssb_periods(num_samples);
81  buff.clear();

83 
84  complex<float> j(0, 1);
85  // Get samples from RF layer and put them in buff variable
86  time_first_pss = chrono::high_resolution_clock::now();
87  try {
88  rf_device->get_samples(&buff_2_ssb_periods, time_first_sample);
89  }catch (const exception& e) {

91  }

93  int cp_lengths_pbch[num_symbols_per_subframe_pbch];
94  int cum_sum_pbch[num_symbols_per_subframe_pbch];
95 
96  free5GRAN::phy::signal_processing::compute_cp_lengths((int) scs/1e3, fft_size, 0, num_symbols_per_subframe_pbch, cp_lengths_pbch, cum_sum_pbch);
97  /*
98  * Take second symbol CP as common CP as SSB is never transmitted at long CP symbols (long CP are transmitted every 0.5ms)
99  */
100  common_cp_length = cp_lengths_pbch[1];
101  /*
102  * Extract first half of buffer (= 1 SSB period)
103  */

113  /*
114  * Computing symbol length and first sample index of PSS in buff
115  */
116  int symbol_duration = fft_size + common_cp_length;
117  int pss_start_index = synchronisation_index - symbol_duration + 1;
118  int sss_init_index = pss_start_index + 2 * symbol_duration + common_cp_length; // = (synchronisation_index - symbol_duration + 1) + 2 * symbol_duration;
119  /*

121  */

134  }

137  */

140  }

143  int n_id_1;
144  float peak_value_sss;

148  */

153  BOOST_LOG_TRIVIAL(trace) << "PCI : "+ to_string(pci);
154 
155 
156  /*

158  */

160  int second_pss_index = pss_start_index + num_samples / 2;
161  int n_id_1_2, n_id_2_2, sync_index_pss_2;
162  float peak_value_pss_2;
163  for (int i = 0; i < fft_size + common_cp_length; i++){
164  second_pss[i] = buff_2_ssb_periods[i + second_pss_index];
165  }
166  /*
167  * Retrieve N ID 2 value from second SSB
168  */
169  free5GRAN::phy::synchronization::search_pss(n_id_2_2,sync_index_pss_2,peak_value_pss_2, common_cp_length, second_pss, fft_size);
170 
171  /*
172  * Extract SSS symbol from second SSB
173  */
174  for (int i = 0; i < fft_size; i++){
175  second_sss[i] = buff_2_ssb_periods[i + second_pss_index + 2 * symbol_duration + common_cp_length];
176  }
177 
178  free5GRAN::phy::synchronization::get_sss(n_id_1_2, peak_value_sss, second_sss, fft_size, n_id_2_2);
179 
180  if (3 * n_id_1_2 + n_id_2_2 == pci){
181  return 0;
182  }else {
183  return 1;
184  }

206  BOOST_LOG_TRIVIAL(trace) << "Extracting PBCH";
207  // Get at least 30ms of signal (=3 frames, at least 2 complete ones)
208  size_t num_samples = max(0.03, ssb_period) * rf_device->getSampleRate();
209  //vector<complex<float>> buff(num_samples);
210  buff.clear();
211  buff.resize(num_samples);
212 
213  double second_frame_time;
214  // Getting samples
215  auto now = chrono::high_resolution_clock::now();
216  try {
217  rf_device->get_samples(&buff, second_frame_time);
218  }catch (const exception& e) {
219  return 1;
220  }
221 
222  /*
223  * SYNCHRONIZING IN THE NEW RECEIVED FRAME
224  * Computing approximate PSS index inside the received buffer using the time reference of the first PSS index SSB initial search
225  */
226  auto time_window = chrono::duration_cast<chrono::microseconds>(now - time_first_pss);
227  int offset_to_ssb_period = (int)(time_window.count() - index_first_pss / (128*scs *1e-6)) % ((int) (ssb_period * 1e6));
228  int index_second_pss = (ssb_period * 1e6 - offset_to_ssb_period) * rf_device->getSampleRate() * 1e-6;
229 
230  ofstream data;
231  data.open("data.txt");
232  for (int i = 0; i < num_samples; i ++){
233  data << buff[i];
234  data << "\n";
235  }
236  data.close();
237 
238  /*
239  * Compute PBCH CP length
240  */
241  int num_symbols_per_subframe_pbch = free5GRAN::NUMBER_SYMBOLS_PER_SLOT_NORMAL_CP * scs/15e3;
242  int cp_lengths_pbch[num_symbols_per_subframe_pbch];
243  int cum_sum_pbch[num_symbols_per_subframe_pbch];
244 
245  free5GRAN::phy::signal_processing::compute_cp_lengths((int) scs/1e3, fft_size, 0, num_symbols_per_subframe_pbch, cp_lengths_pbch, cum_sum_pbch);
246  common_cp_length = cp_lengths_pbch[1];
247  int symbol_duration = fft_size + common_cp_length;
248 
249  vector<complex<float>> pss_signal(48 * symbol_duration);
250 
251  /*
252  * Isolate PSS signal around calculated new PSS occurence (based on timestamp of first synchronization step)
253  */
254  int begin_offset, end_offset;
255  if (pss_signal.size()/2 <= index_second_pss && pss_signal.size()/2 <= num_samples - index_second_pss){
256  begin_offset = pss_signal.size()/2;
257  end_offset = pss_signal.size()/2;
258  } else if (pss_signal.size()/2 > index_second_pss){
259  begin_offset = index_second_pss;
260  end_offset = pss_signal.size() - index_second_pss;
261  } else if (pss_signal.size()/2 > num_samples - index_second_pss) {
262  end_offset = num_samples - index_second_pss;
263  begin_offset = pss_signal.size() - (num_samples - index_second_pss);
264  }
265 
266  // Extracting the signal around the PSS approximation
267  int count = 0;
268  for (int i = -begin_offset; i < end_offset; i ++){
269  pss_signal[count] = buff[(int) index_second_pss + i];
270  count ++;
271  }
272 
273  int synchronisation_index;
274  float peak_value;
275 
276  /*
277  * Downsample signal for better performance
278  */
279  int downsampling_factor = fft_size / free5GRAN::PSS_SSS_FFT_SIZE;
280  BOOST_LOG_TRIVIAL(trace) << "PSS synchronization downsampling factor: " << downsampling_factor;
281  int symbol_duration_downsampled = symbol_duration / downsampling_factor;
282 
283  vector<complex<float>> pss_signal_downsampled(48 * symbol_duration_downsampled);
284  for (int i = 0; i < 48 * symbol_duration_downsampled; i ++){
285  pss_signal_downsampled[i] = pss_signal[downsampling_factor * i];
286  }
287 
288  free5GRAN::phy::synchronization::search_pss(this->n_id_2,synchronisation_index,peak_value, common_cp_length / downsampling_factor, pss_signal_downsampled,fft_size / downsampling_factor);
289 
290  int pss_start_index = downsampling_factor * (synchronisation_index - symbol_duration_downsampled + 1);
291 
292  /*
293  * Once synchronization is made on downsampled signal, it can be performed on full signal for finer results
294  */
295  vector<complex<float>> fine_pss_signal(symbol_duration + (2 * downsampling_factor + 1));
296  count = 0;
297  for (int i = pss_start_index - downsampling_factor; i < pss_start_index + symbol_duration + (downsampling_factor + 1); i ++){
298  fine_pss_signal[count] = pss_signal[i];
299  count ++;
300  }
301  free5GRAN::phy::synchronization::search_pss(this->n_id_2,synchronisation_index,peak_value, common_cp_length, fine_pss_signal,fft_size);
302 
303  pss_start_index = pss_start_index + (synchronisation_index - symbol_duration + 1 - downsampling_factor);
304  int buffer_pss_index = pss_start_index + index_second_pss - begin_offset;
305  index_first_pss = buffer_pss_index;
306  int sss_init_index = buffer_pss_index + 2 * symbol_duration + common_cp_length;
307 
308  vector<complex<float>> sss_signal(fft_size);
309  /*
310  * Extracting the SSS signal based on sss_init_index and common_cp_length
311  */
312  for (int i = 0; i < fft_size; i++){
313  sss_signal[i] = buff[i + sss_init_index];
314  }
315 
316  float peak_value_sss;
317  /*
318  * Get SSS correlation result
319  */
320  free5GRAN::phy::synchronization::get_sss(this->n_id_1, peak_value_sss, sss_signal,fft_size,this->n_id_2);
321  if (pci == 3 * n_id_1 + n_id_2){
322  BOOST_LOG_TRIVIAL(trace) << "PCI confirmed";
323  cell_confirmed = true;
324  }else{
325  BOOST_LOG_TRIVIAL(trace) << "PCI not confirmed";
326  cell_confirmed = false;
327  return 1;
328  }
329  /*
330  * WE ARE NOW SYNCHONIZED IN OUR NEW FRAME
331  * Trying to extract DMRS AND PBCH
332  */
333 
334  vector<complex<float>> ssb_signal(4 * symbol_duration), new_ssb_signal(4 * symbol_duration);
335 
336  // Extract SSB signal
337  for (int i = 0; i < free5GRAN::NUM_SYMBOLS_SSB * symbol_duration; i ++){
338  ssb_signal[i] = buff[i + buffer_pss_index];
339  }
340 
341  /*
342  * Fine frequency correlation
343  * Getting phase offset between CP and corresponding part of the OFDM symbol for each of the 4 symbols.
344  * phase_offset is the mean phase offset
345  */
346  free5GRAN::phy::signal_processing::compute_fine_frequency_offset(ssb_signal, symbol_duration, fft_size, common_cp_length, scs, freq_offset, free5GRAN::NUM_SYMBOLS_SSB);
347 
348  // Correcting signal based on frequency offset
349  free5GRAN::phy::signal_processing::transpose_signal(&buff, freq_offset, rf_device->getSampleRate(), buff.size());
350 
351  vector<complex<float>> pbch_modulation_symbols(free5GRAN::SIZE_SSB_PBCH_SYMBOLS), final_pbch_modulation_symbols(free5GRAN::SIZE_SSB_PBCH_SYMBOLS);
352 
353  /*
354  * Extracting DMRS AND PBCH modulation symbols
355  * ref is the reference grid for resource element demapper
356  */
357  vector<complex<float>> temp_mod_symbols, temp_mod_symbols2,temp_mod_symbols_dmrs;
358 
359  complex<float> pbch_symbols[free5GRAN::SIZE_SSB_PBCH_SYMBOLS];
360  complex<float> dmrs_symbols[free5GRAN::SIZE_SSB_DMRS_SYMBOLS];
361  complex<float> sss_symbols[free5GRAN::SIZE_PSS_SSS_SIGNAL];
362  /*
363  * ref[0] -> indexes of PBCH resource elements
364  * ref[1] -> indexes of DMRS resource elements
365  */
366  vector<vector<vector<int>>> ref(3, vector<vector<int>>(free5GRAN::SIZE_SSB_DMRS_SYMBOLS, vector<int>(free5GRAN::NUM_SC_SSB)));
367  /*
368  * channel_indexes[0] contains PBCH samples indexes
369  * channel_indexes[1] contains DMRS samples indexes
370  * channel_indexes[2] contains SSS samples indexes
371  */
372  vector<vector<vector<int>>> channel_indexes = {vector<vector<int>>(2, vector<int>(free5GRAN::SIZE_SSB_PBCH_SYMBOLS)), vector<vector<int>>(2, vector<int>(free5GRAN::SIZE_SSB_DMRS_SYMBOLS)), vector<vector<int>>(2, vector<int>(free5GRAN::SIZE_PSS_SSS_SIGNAL))};
373 
374  vector<vector<complex<float>>> ssb_symbols(free5GRAN::NUM_SYMBOLS_SSB - 1, vector<complex<float>>(free5GRAN::NUM_SC_SSB));
375 
376  int cum_sum_fft[free5GRAN::NUM_SYMBOLS_SSB];
377  for (int symbol = 0; symbol < free5GRAN::NUM_SYMBOLS_SSB; symbol ++){
378  cum_sum_fft[symbol] = symbol * symbol_duration;
379  }
380 
381  /*
382  * Recover RE grid from time domain signal
383  */
384  free5GRAN::phy::signal_processing::fft(ssb_signal, ssb_symbols,fft_size,cp_lengths_pbch,&cum_sum_fft[0],free5GRAN::NUM_SYMBOLS_SSB - 1,free5GRAN::NUM_SC_SSB,1,0);
385 
386  free5GRAN::phy::physical_channel::compute_pbch_indexes(ref, pci);
387  /*
388  * Channel demapping using computed ref grid
389  */
390  complex<float>* output_channels[] = {pbch_symbols, dmrs_symbols, sss_symbols};
391  free5GRAN::phy::signal_processing::channel_demapper(ssb_symbols, ref, output_channels, channel_indexes, 3, free5GRAN::NUM_SYMBOL_PBCH_SSB, free5GRAN::NUM_SC_SSB);
392 
393  /*
394  * Channel estimation and equalization
395  * Creating coefficients arrays
396  */
397  vector<vector<vector<complex<float>>>> coefficients(free5GRAN::MAX_I_BAR_SSB, vector<vector<complex<float>>>(free5GRAN::NUM_SYMBOL_PBCH_SSB, vector<complex<float>>(free5GRAN::NUM_SC_SSB)));
398 
399 
400  complex<float> dmrs_sequence[free5GRAN::SIZE_SSB_DMRS_SYMBOLS];
401  float snr[free5GRAN::MAX_I_BAR_SSB];
402 
403  /*
404  * For each possible iBarSSB value, estimate the corresponding transport_channel
405  */
406  for (int i = 0; i < free5GRAN::MAX_I_BAR_SSB; i ++){
407  free5GRAN::utils::sequence_generator::generate_pbch_dmrs_sequence(pci,i,dmrs_sequence);
408  free5GRAN::phy::signal_processing::channelEstimation(dmrs_symbols, dmrs_sequence, channel_indexes[1],coefficients[i], snr[i], free5GRAN::NUM_SC_SSB, free5GRAN::NUM_SYMBOL_PBCH_SSB , free5GRAN::SIZE_SSB_DMRS_SYMBOLS);
409  }
410  /*
411  * Choose the iBarSSB value that maximizes the SNR
412  */
413  max_snr = snr[0];

415  for (int i = 1; i < free5GRAN::MAX_I_BAR_SSB ; i ++){

425  }
426 
427  ss_pwr.ss_rsrp = 0;
428  for (int i = 0; i < free5GRAN::SIZE_PSS_SSS_SIGNAL; i ++){

430  }
431  for (int i = 0; i < free5GRAN::SIZE_SSB_DMRS_SYMBOLS; i ++){
432  ss_pwr.ss_rsrp +=pow(abs(dmrs_symbols[i]),2);
433  }
434  ss_pwr.ss_rsrp /= (free5GRAN::SIZE_PSS_SSS_SIGNAL + free5GRAN::SIZE_SSB_DMRS_SYMBOLS);
435 
436  ss_pwr.ss_rssi = 0;
437  for (int symb = 0 ; symb < free5GRAN::NUM_SYMBOLS_SSB - 1; symb++){
438  for (int sc = 0; sc < free5GRAN::NUM_SC_SSB; sc ++){
439  ss_pwr.ss_rssi += pow(abs(ssb_symbols[symb][sc]),2);
440  }
441  }
442  // 20 is the number of RB in SSB block
443  int n_rb = 20;
444  ss_pwr.ss_rssi /= n_rb;
445  ss_pwr.ss_rsrq = 10 * log(n_rb * ss_pwr.ss_rsrp / ss_pwr.ss_rssi);
446  // Converting RSRP and RSSI from W to dBm
447  ss_pwr.ss_rsrp = 10 * log10(ss_pwr.ss_rsrp) + 30;
448  ss_pwr.ss_rssi = 10 * log10(ss_pwr.ss_rssi) + 30;
449  ss_pwr.ss_sinr = max_snr;
450 
451  this->i_b_ssb = i_b_ssb;
452  if (l_max == 4){
453  this-> i_ssb = i_b_ssb % 4;
454  }else {
455  this-> i_ssb = i_b_ssb;
456  }
457  this->pci = pci;
458 
459  /*
460  * Physical and transport channel decoding
461  * MIB parsing
462  */
463  int bch_bits[free5GRAN::SIZE_SSB_PBCH_SYMBOLS * 2];
464  free5GRAN::phy::physical_channel::decode_pbch(final_pbch_modulation_symbols, i_ssb, pci, bch_bits);
465  int mib_bits[free5GRAN::BCH_PAYLOAD_SIZE];
466  free5GRAN::phy::transport_channel::decode_bch(bch_bits, crc_validated, mib_bits, pci);
467  free5GRAN::utils::common_utils::parse_mib(mib_bits, mib_object);
468  return 0;
469 }

514 void phy::search_pdcch(bool &dci_found) {
537  /*
538  * If SSB offset is greater than 23, PDCCH is not present in the current BWP
539  */
540  if(mib_object.k_ssb > 23){
541  dci_found = false;
542  return;

685  */

732  */

780  */

786  }
787  }
788  /*
789  * Channel estimation
790  */
791  free5GRAN::phy::signal_processing::channelEstimation(dmrs_symbols, dmrs_sequence, channel_indexes[1],coefficients, snr, 12 * pdcch_ss_mon_occ.n_rb_coreset, pdcch_ss_mon_occ.n_symb_coreset , agg_level * free5GRAN::NUMBER_REG_PER_CCE * 3);
792  /*
793  * Channel equalization
794  */
795  for (int sc = 0; sc < agg_level * free5GRAN::NUMBER_REG_PER_CCE * 9; sc ++){
796  pdcch_symbols[sc] = (temp_pdcch_symbols[sc]) * conj(coefficients[channel_indexes[0][0][sc]][channel_indexes[0][1][sc]]) / (float) pow(abs(coefficients[channel_indexes[0][0][sc]][channel_indexes[0][1][sc]]),2);
797  }
798  /*
799  * PDCCH and DCI decoding
800  */
801  free5GRAN::phy::physical_channel::decode_pdcch(pdcch_symbols,dci_bits,agg_level, reg_bundles_ns, reg_bundles, pci);
802  free5GRAN::phy::transport_channel::decode_dci(dci_bits, agg_level * free5GRAN::NUMBER_REG_PER_CCE * 9 * 2, K, free5GRAN::SI_RNTI, validated, dci_decoded_bits);
803  /*
804  * If DCI CRC is validated, candidate is validated, blind search ends
805  */
806  if (validated){
807  data_pdcch.open("output_files/pdcch_constellation.txt");
808  for (int sc = 0; sc < agg_level * free5GRAN::NUMBER_REG_PER_CCE * 9; sc ++){
809  data_pdcch << pdcch_symbols[sc];
810  data_pdcch << "\n";
811  }
812  data_pdcch.close();
813  goto dci_found_and_validated;
814  }
815 
816  }
817  }else {
818  break;
819  }
820  }
821  }
822 
823  dci_found_and_validated:
824  BOOST_LOG_TRIVIAL(trace) << "## DCI FOUND AND " << ((validated) ? "VALIDATED" : "NOT VALIDATED");
825 
826  dci_found = false;
827  if (validated){
828  parse_dci_1_0_si_rnti(dci_decoded_bits,freq_domain_ra_size,dci_1_0_si_rnti);
829  n_size_bwp = pdcch_ss_mon_occ.n_rb_coreset;
830  /*
831  * In current version, only redundancy version 0 and 3 are supported for DL-SCH decoding
832  */
833  if (dci_1_0_si_rnti.rv == 0 || dci_1_0_si_rnti.rv == 3){
834  dci_found = true;
835  }
